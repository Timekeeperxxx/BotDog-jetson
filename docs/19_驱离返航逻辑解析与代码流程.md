```python
async def _on_returning(self, detections, frame):
    """视觉闭环返航：朝起始位置后退，面积和中心都回到起点后蹲坐完成任务。"""
    
    # 超时保护
    elapsed = time.monotonic() - self._return_start_time
    if elapsed >= self._config.GUARD_RETURN_DURATION_S * 3:
        await self._send_command_safe("stop")
        await self._send_command_safe("sit")
        self._finish_mission_and_reset(frame)
        return
    
    zone = self._current_zone
    
    # 区域丢失保护 → 盲退
    if zone is None or self._start_zone is None:
        await self._send_command_safe("backward")
        return
    
    # 调用视觉伺服控制器计算返航指令
    cmd, is_returned = self._servo.compute_returning(
        curr_bbox=zone.bbox,
        start_bbox=self._start_zone.bbox,           # ← 对标起始区域
        frame_width=self._frame_width,
        pos_tolerance_px=self._config.GUARD_RETURN_POS_TOLERANCE_PX,
        area_tolerance_ratio=self._config.GUARD_RETURN_AREA_TOLERANCE_RATIO,
    )
    await self._send_command_safe(cmd)
    
    # 连续 N 帧满足到位条件才确认完成
    if is_returned:
        self._return_stable_counter += 1
        if self._return_stable_counter >= self._return_stable_threshold:
            await self._send_command_safe("stop")
            await self._send_command_safe("sit")
            self._finish_mission_and_reset(frame)
```

**关键成员变量**：
- `self._start_zone: Optional[ZoneDetection]` (L94)：记录驱离开始时的区域快照
- `self._return_start_time: float` (L77)：返航开始时间戳
- `self._return_stable_counter: int` (L96)：返航稳定帧数计数
- `self._return_stable_threshold: int` (L97)：返航完成阈值（默认15帧）

### 2️⃣ **backend/visual_servo_controller.py**（核心算法）

**职责**：视觉伺服计算，生成驱离/返航指令

#### 前进伺服 `compute_advancing()`

```python
def compute_advancing(self, 
                      curr_bbox: Tuple[int, int, int, int], 
                      frame_width: int, 
                      frame_height: int, 
                      max_view_ratio: float, 
                      edge_margin_ratio: float = 0.08
                      ) -> Tuple[str, bool]:
    """
    前进阶段指令计算
    :return: (指令, 是否到达了极限应该停车)
    """
    x, y, w, h = curr_bbox
    center_x = x + w // 2
    
    # 贴脸保护：宽或高任意一边触达阈值 → 停车
    if w >= frame_width * max_view_ratio or h >= frame_height * max_view_ratio:
        return "stop", True
    
    # 偏航纠正
    error_x = center_x - (frame_width // 2)
    if abs(error_x) > self._yaw_deadband_px:
        cmd = "left" if error_x < 0 else "right"
        return cmd, False
    
    # 边缘裕量保护
    margin_x = int(frame_width * edge_margin_ratio)
    margin_y = int(frame_height * edge_margin_ratio)
    too_close_to_edge = (
        x < margin_x or 
        (x + w) > (frame_width - margin_x) or 
        (y + h) > (frame_height - margin_y)
    )
    if too_close_to_edge:
        return "stop", False
    
    return "forward", False
```

#### 返航伺服 `compute_returning()`（最关键）

```python
def compute_returning(self, 
                      curr_bbox: Tuple[int, int, int, int],
                      start_bbox: Tuple[int, int, int, int],    # ← 起始状态
                      frame_width: int, 
                      pos_tolerance_px: int, 
                      area_tolerance_ratio: float
                      ) -> Tuple[str, bool]:
    """
    返航阶段指令计算
    :param curr_bbox: 当前锚点 (x, y, w, h)
    :param start_bbox: 起始锚点（目标状态）
    :param pos_tolerance_px: 位置允许误差（像素）
    :param area_tolerance_ratio: 面积允许误差（比例）
    :return: (指令, 是否已回到起始状态)
    """
    curr_x, curr_y, curr_w, curr_h = curr_bbox
    start_x, start_y, start_w, start_h = start_bbox
    
    curr_center_x = curr_x + curr_w // 2
    start_center_x = start_x + start_w // 2
    
    # 保证不是"斜着倒车退走"
    error_x = curr_center_x - (frame_width // 2)
    cmd = "backward"
    
    if abs(error_x) > self._yaw_deadband_px:
        if error_x < 0:
            cmd = "left"
        else:
            cmd = "right"
    
    # 判定返航完成的两个标准
    # 标准 1：面积缩小到起步时的设定值
    curr_area = curr_w * curr_h
    start_area = start_w * start_h
    area_upper_bound = start_area * (1.0 + area_tolerance_ratio)
    has_reached_z = (curr_area <= area_upper_bound)
    
    # 标准 2：位置归中
    has_reached_x = (abs(curr_center_x - start_center_x) <= pos_tolerance_px)
    
    if has_reached_z and has_reached_x:
        is_returned = True
        cmd = "stop"
    else:
        is_returned = False
    
    return cmd, is_returned
```

**关键判定逻辑**：
1. **面积判定**：`curr_area <= start_area * (1 + 0.15)`
   - 代表狗后退到接近原始距离
2. **位置判定**：`|curr_center_x - start_center_x| <= 30px`
   - 代表锚点中心回到水平中位线附近
3. **同时满足两者** → 返航完成

### 4️⃣ **backend/yellow_zone_detector.py**（阶段1：区域检测）

**职责**：实现纯视觉区域识别，输出动态 ZoneDetection

```python
@dataclass
class ZoneDetection:
    polygon: np.ndarray               # shape (4, 2)，四边形顶点
    bbox: Tuple[int, int, int, int]   # (x, y, w, h)
    center: Tuple[int, int]           # 中心点 (cx, cy)
    area: float                       # 像素面积
    angle: float                      # 旋转角度
    quality: float                    # 0.0-1.0 综合质量分
```

**检测管线**：
1. BGR → HSV 转换
2. 黄色阈值分割 (H:20-35)
3. 形态学处理（闭运算 + 开运算）
4. 轮廓提取与四边形拟合
5. 黑边验证（防止误识别）
6. 输出 ZoneDetection 或 None
